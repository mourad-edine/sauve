// components/gallery/GalleryCarousel.js
'use client';

import { useState, useRef, useEffect } from 'react';
import { FiZoomIn, FiChevronLeft, FiChevronRight, FiGrid, FiX } from 'react-icons/fi';
import { motion, AnimatePresence } from 'framer-motion';

export default function GalleryCarousel({ images, onImageClick }) {
  const [currentIndex, setCurrentIndex] = useState(0);
  const [isFullscreen, setIsFullscreen] = useState(false);
  const [isPlaying, setIsPlaying] = useState(false);
  const [hoveredId, setHoveredId] = useState(null);
  const carouselRef = useRef(null);
  const autoPlayRef = useRef(null);

  // Navigation
  const goToSlide = (index) => {
    setCurrentIndex(index);
  };

  const goToPrev = () => {
    setCurrentIndex((prev) => (prev === 0 ? images.length - 1 : prev - 1));
  };

  const goToNext = () => {
    setCurrentIndex((prev) => (prev === images.length - 1 ? 0 : prev + 1));
  };

  // Auto-play
  useEffect(() => {
    if (isPlaying && images.length > 1) {
      autoPlayRef.current = setInterval(goToNext, 3000);
    }
    return () => clearInterval(autoPlayRef.current);
  }, [isPlaying, images.length]);

  // Keyboard navigation
  useEffect(() => {
    const handleKeyDown = (e) => {
      if (isFullscreen) {
        if (e.key === 'Escape') setIsFullscreen(false);
        if (e.key === 'ArrowLeft') goToPrev();
        if (e.key === 'ArrowRight') goToNext();
        if (e.key === ' ') {
          e.preventDefault();
          setIsPlaying(!isPlaying);
        }
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [isFullscreen, isPlaying]);

  if (!images?.length) {
    return (
      <div className="text-center py-16 text-gray-500">
        <p className="text-lg font-medium">Aucune image disponible</p>
      </div>
    );
  }

  return (
    <div className="relative">
      {/* Carousel principal */}
      <div className="relative overflow-hidden bg-white" ref={carouselRef}>
        {/* Image principale */}
        <div className="relative aspect-[16/9] bg-gray-50 overflow-hidden">
          <AnimatePresence mode="wait">
            <motion.div
              key={currentIndex}
              initial={{ opacity: 0, scale: 0.95 }}
              animate={{ opacity: 1, scale: 1 }}
              exit={{ opacity: 0, scale: 1.05 }}
              transition={{ duration: 0.4 }}
              className="absolute inset-0"
            >
              <img
                src={images[currentIndex].imageUrl}
                alt={images[currentIndex].title || 'Photo'}
                className="w-full h-full object-cover cursor-pointer"
                onClick={() => onImageClick(currentIndex)}
              />
              
              {/* Overlay info */}
              <div className="absolute bottom-0 left-0 right-0 bg-gradient-to-t from-black/60 to-transparent p-6">
                <div className="flex items-center justify-between">
                  <div className="text-white">
                    <h3 className="text-xl font-medium mb-1">
                      {images[currentIndex].title || 'Sans titre'}
                    </h3>
                    {images[currentIndex].description && (
                      <p className="text-sm opacity-90">
                        {images[currentIndex].description}
                      </p>
                    )}
                  </div>
                  <div className="flex items-center gap-3">
                    <button
                      onClick={() => onImageClick(currentIndex)}
                      className="p-3 bg-white/20 backdrop-blur-sm rounded-full hover:bg-white/30 transition-colors"
                    >
                      <FiZoomIn className="w-5 h-5 text-white" />
                    </button>
                    <span className="text-sm text-white/80">
                      {currentIndex + 1} / {images.length}
                    </span>
                  </div>
                </div>
              </div>
            </motion.div>
          </AnimatePresence>
        </div>

        {/* Navigation */}
        {images.length > 1 && (
          <>
            <button
              onClick={goToPrev}
              className="absolute left-4 top-1/2 -translate-y-1/2 p-3 bg-white/90 backdrop-blur-sm rounded-full shadow-lg hover:bg-white transition-all hover:scale-105"
            >
              <FiChevronLeft className="w-5 h-5 text-gray-800" />
            </button>
            
            <button
              onClick={goToNext}
              className="absolute right-4 top-1/2 -translate-y-1/2 p-3 bg-white/90 backdrop-blur-sm rounded-full shadow-lg hover:bg-white transition-all hover:scale-105"
            >
              <FiChevronRight className="w-5 h-5 text-gray-800" />
            </button>
          </>
        )}

        {/* Controls */}
        <div className="absolute bottom-4 left-1/2 -translate-x-1/2 flex items-center gap-3 bg-white/90 backdrop-blur-sm px-4 py-2 rounded-full shadow-lg">
          <button
            onClick={() => setIsFullscreen(!isFullscreen)}
            className="p-2 hover:bg-gray-100 rounded transition-colors"
            title="Plein écran"
          >
            <FiGrid className="w-4 h-4 text-gray-700" />
          </button>
          
          <button
            onClick={() => setIsPlaying(!isPlaying)}
            className="p-2 hover:bg-gray-100 rounded transition-colors"
            title={isPlaying ? 'Pause' : 'Lecture automatique'}
          >
            <div className={`w-4 h-4 border-2 border-gray-700 ${isPlaying ? 'bg-gray-700' : ''}`} />
          </button>
          
          {/* Dots navigation */}
          <div className="flex items-center gap-2 mx-2">
            {images.map((_, idx) => (
              <button
                key={idx}
                onClick={() => goToSlide(idx)}
                className={`w-2 h-2 rounded-full transition-all ${
                  idx === currentIndex 
                    ? 'bg-gray-800 w-6' 
                    : 'bg-gray-400 hover:bg-gray-600'
                }`}
              />
            ))}
          </div>
        </div>
      </div>

      {/* Miniatures horizontales */}
      <div className="mt-6 px-8">
        <div className="flex gap-3 overflow-x-auto pb-4 scrollbar-hide">
          {images.map((image, idx) => (
            <motion.div
              key={image.id}
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ delay: idx * 0.05 }}
              className={`flex-shrink-0 relative cursor-pointer transition-all ${
                idx === currentIndex 
                  ? 'ring-2 ring-gray-900 scale-105' 
                  : 'hover:opacity-80'
              }`}
              onClick={() => goToSlide(idx)}
              onMouseEnter={() => setHoveredId(image.id)}
              onMouseLeave={() => setHoveredId(null)}
            >
              <div className="relative aspect-[4/3] w-24 overflow-hidden bg-gray-100">
                <img
                  src={image.imageUrl}
                  alt={image.title || 'Miniature'}
                  className="w-full h-full object-cover"
                />
                {hoveredId === image.id && idx !== currentIndex && (
                  <div className="absolute inset-0 bg-black/20 flex items-center justify-center">
                    <div className="p-1.5 bg-white/90 rounded">
                      <FiChevronRight className="w-3 h-3 text-gray-800" />
                    </div>
                  </div>
                )}
              </div>
              {idx === currentIndex && (
                <div className="absolute -bottom-1 left-1/2 -translate-x-1/2 w-6 h-1 bg-gray-900" />
              )}
            </motion.div>
          ))}
        </div>
      </div>

      {/* Modal plein écran */}
      <AnimatePresence>
        {isFullscreen && (
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            className="fixed inset-0 z-50 bg-black flex flex-col"
          >
            {/* Header */}
            <div className="absolute top-0 left-0 right-0 p-6 flex items-center justify-between z-10 bg-gradient-to-b from-black/50 to-transparent">
              <div className="text-white">
                <h2 className="text-xl font-medium">{images[currentIndex].title}</h2>
                {images[currentIndex].description && (
                  <p className="text-sm opacity-80 mt-1">
                    {images[currentIndex].description}
                  </p>
                )}
              </div>
              
              <div className="flex items-center gap-4">
                <button
                  onClick={() => setIsPlaying(!isPlaying)}
                  className="text-white hover:bg-white/20 p-2 rounded"
                >
                  {isPlaying ? '❚❚' : '▶'}
                </button>
                <button
                  onClick={() => setIsFullscreen(false)}
                  className="text-white hover:bg-white/20 p-2 rounded"
                >
                  <FiX className="w-6 h-6" />
                </button>
              </div>
            </div>

            {/* Image plein écran */}
            <div className="flex-1 flex items-center justify-center p-8">
              <motion.img
                key={currentIndex}
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                exit={{ opacity: 0 }}
                src={images[currentIndex].imageUrl}
                alt={images[currentIndex].title}
                className="max-w-full max-h-full object-contain"
              />
            </div>

            {/* Navigation plein écran */}
            <div className="absolute bottom-0 left-0 right-0 p-6 flex items-center justify-between z-10 bg-gradient-to-t from-black/50 to-transparent">
              <button
                onClick={goToPrev}
                className="p-3 text-white hover:bg-white/20 rounded-full transition-colors"
              >
                <FiChevronLeft className="w-6 h-6" />
              </button>
              
              <div className="flex items-center gap-3">
                {images.map((_, idx) => (
                  <button
                    key={idx}
                    onClick={() => goToSlide(idx)}
                    className={`w-2 h-2 rounded-full ${
                      idx === currentIndex ? 'bg-white' : 'bg-white/50'
                    }`}
                  />
                ))}
              </div>
              
              <button
                onClick={goToNext}
                className="p-3 text-white hover:bg-white/20 rounded-full transition-colors"
              >
                <FiChevronRight className="w-6 h-6" />
              </button>
            </div>

            {/* Miniatures bas plein écran */}
            <div className="absolute bottom-20 left-0 right-0">
              <div className="flex justify-center gap-2 px-8">
                {images.map((img, idx) => (
                  <button
                    key={img.id}
                    onClick={() => goToSlide(idx)}
                    className={`flex-shrink-0 w-16 h-12 overflow-hidden ${
                      idx === currentIndex ? 'ring-2 ring-white' : 'opacity-60 hover:opacity-100'
                    }`}
                  >
                    <img
                      src={img.imageUrl}
                      alt=""
                      className="w-full h-full object-cover"
                    />
                  </button>
                ))}
              </div>
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
